//gets triangle number of a given val 
int triangle(int val)
{
  return ((pow(val,2)+val)/2);
}

int numDivisors(int num)
{
  int count = 0;
  if(sqrt(num)*sqrt(num)==num)
  {
    count = count -1;
  }
  for(int i = 1;i<=sqrt(num);i++)
  {
    if(num%i==0)
    {
      count = count + 2;
    }
  }
  return count;
}//-1
